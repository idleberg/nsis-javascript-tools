messagebox MB_OK "Hello"

SETOUTPATH $INSTDIR

Script(

DetailPRINT "test"

==>

Script(
  Command(CommandName, String))

# Case-insensitive function block

function .onInit
  messagebox MB_OK "Hello"
functionend

==>

Script(
  Block(FunctionBlock(
    Function, Identifier,
    Command(CommandName, Constant, String),
    FunctionEnd)))

section "Install"
  setoutpath $INSTDIR
sectionend

    SectionEnd)))

    Command(CommandName, ArgumentValue),
